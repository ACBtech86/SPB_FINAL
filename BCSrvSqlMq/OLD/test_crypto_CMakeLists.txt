cmake_minimum_required(VERSION 3.20)
project(CryptoTest)

# Set C++17
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find OpenSSL
set(CMAKE_PREFIX_PATH "C:/vcpkg/installed/x64-windows")
find_package(OpenSSL REQUIRED)

# Create executable
add_executable(test_crypto_full test_crypto_full.cpp)

# Link OpenSSL
target_link_libraries(test_crypto_full PRIVATE OpenSSL::SSL OpenSSL::Crypto)

# Include directories
target_include_directories(test_crypto_full PRIVATE ${OPENSSL_INCLUDE_DIR})

# Copy OpenSSL DLLs to output directory
if(WIN32)
    add_custom_command(TARGET test_crypto_full POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "C:/vcpkg/installed/x64-windows/bin/libcrypto-3-x64.dll"
        "C:/vcpkg/installed/x64-windows/bin/libssl-3-x64.dll"
        $<TARGET_FILE_DIR:test_crypto_full>
    )
endif()
